17   package com.example.android.todolist; 
18    
19   import android.content.Intent; 
20   import android.os.Bundle; 
21   import android.support.v7.app.AppCompatActivity; 
22   import android.view.View; 
23   import android.widget.Button; 
24   import android.widget.EditText; 
25   import android.widget.RadioGroup; 
26    
27   import com.example.android.todolist.database.AppDatabase; 
28   import com.example.android.todolist.database.TaskEntry; 
29    
30   import java.util.Date; 
31    
32    
33   public class AddTaskActivity extends AppCompatActivity { 
34    
35       // Extra for the task ID to be received in the intent 
36       public static final String EXTRA_TASK_ID = "extraTaskId"; 
37       // Extra for the task ID to be received after rotation 
38       public static final String INSTANCE_TASK_ID = "instanceTaskId"; 
39       // Constants for priority 
40       public static final int PRIORITY_HIGH = 1; 
41       public static final int PRIORITY_MEDIUM = 2; 
42       public static final int PRIORITY_LOW = 3; 
43       // Constant for default task id to be used when not in update mode 
44       private static final int DEFAULT_TASK_ID = -1; 
45       // Constant for logging 
46       private static final String TAG = AddTaskActivity.class.getSimpleName(); 
47       // Fields for views 
48       EditText mEditText; 
49       RadioGroup mRadioGroup; 
50       Button mButton; 
51    
52       private int mTaskId = DEFAULT_TASK_ID; 
53    
54       // Member variable for the Database 
55       private AppDatabase mDb; 
56    
57       protected void onCreate(Bundle savedInstanceState) { 
58           super.onCreate(savedInstanceState); 
59           setContentView(R.layout.activity_add_task); 
60    
61           initViews(); 
62    
63           mDb = AppDatabase.getInstance(getApplicationContext()); 
64    
65           if (savedInstanceState != null && savedInstanceState.containsKey(INSTANCE_TASK_ID)) { 
66               mTaskId = savedInstanceState.getInt(INSTANCE_TASK_ID, DEFAULT_TASK_ID); 
67           } 
68    
69           Intent intent = getIntent(); 
70           if (intent != null && intent.hasExtra(EXTRA_TASK_ID)) { 
71               mButton.setText(R.string.update_button); 
72               if (mTaskId == DEFAULT_TASK_ID) { 
73                   // populate the UI 
74                   // COMPLETED (3) Assign the value of EXTRA_TASK_ID in the intent to mTaskId 
75                   // Use DEFAULT_TASK_ID as the default 
76                   mTaskId = intent.getIntExtra(EXTRA_TASK_ID, -1); 
77                   // COMPLETED (4) Get the diskIO Executor from the instance of AppExecutors and 
78                   // call the diskIO execute method with a new Runnable and implement its run method 
79                   AppExecutors.getInstance().diskIO().execute(new Runnable() { 
80                       @Override 
81                       public void run() { 
82                           // COMPLETED (5) Use the loadTaskById method to retrieve the task with id mTaskId and 
83                           // assign its value to a final TaskEntry variable 
84                           final TaskEntry task = mDb.taskDao().loadTaskById(mTaskId); 
85                           runOnUiThread(new Runnable() { 
86                               @Override 
87                               public void run() { 
88                                   // COMPLETED (6) Call the populateUI method with the retrieve tasks 
89                                   // Remember to wrap it in a call to runOnUiThread 
90                                   populateUI(task); 
91                               } 
92                           }); 
93                       } 
94                   }); 
95    
96               } 
97           } 
98       } 
99    
100      @Override 
101      protected void onSaveInstanceState(Bundle outState) { 
102          outState.putInt(INSTANCE_TASK_ID, mTaskId); 
103          super.onSaveInstanceState(outState); 
104      } 
105   
106      /** 
107       * initViews is called from onCreate to init the member variable views 
108       */ 
109      private void initViews() { 
110          mEditText = findViewById(R.id.editTextTaskDescription); 
111          mRadioGroup = findViewById(R.id.radioGroup); 
112   
113          mButton = findViewById(R.id.saveButton); 
114          mButton.setOnClickListener(new View.OnClickListener() { 
115              @Override 
116              public void onClick(View view) { 
117                  onSaveButtonClicked(); 
118              } 
119          }); 
120      } 
121   
122      /** 
123       * populateUI would be called to populate the UI when in update mode 
124       * 
125       * @param task the taskEntry to populate the UI 
126       */ 
127      private void populateUI(TaskEntry task) { 
128          // COMPLETED (7) return if the task is null 
129          if(task == null) return; 
130          // COMPLETED (8) use the variable task to populate the description and priority 
131          mEditText.setText(task.getDescription()); 
132          setPriorityInViews(task.getPriority()); 
133      } 
134   
135      /** 
136       * onSaveButtonClicked is called when the "save" button is clicked. 
137       * It retrieves user input and inserts that new task data into the underlying database. 
138       */ 
139      public void onSaveButtonClicked() { 
140          String description = mEditText.getText().toString(); 
141          int priority = getPriorityFromViews(); 
142          Date date = new Date(); 
143   
144          final TaskEntry taskEntry = new TaskEntry(description, priority, date); 
145          AppExecutors.getInstance().diskIO().execute(new Runnable() { 
146              @Override 
147              public void run() { 
148                  // COMPLETED (9) insert the task only if mTaskId matches DEFAULT_TASK_ID 
149                  // Otherwise update it 
150                  // call finish in any case 
151                  if(mTaskId == DEFAULT_TASK_ID) { 
152                      mDb.taskDao().insertTask(taskEntry); 
153                  } else { 
154                      taskEntry.setId(mTaskId); 
155                      mDb.taskDao().updateTask(taskEntry); 
156                  } 
157                  finish(); 
158              } 
159          }); 
160      } 
161   
162      /** 
163       * getPriority is called whenever the selected priority needs to be retrieved 
164       */ 
165      public int getPriorityFromViews() { 
166          int priority = 1; 
167          int checkedId = ((RadioGroup) findViewById(R.id.radioGroup)).getCheckedRadioButtonId(); 
168          switch (checkedId) { 
169              case R.id.radButton1: 
170                  priority = PRIORITY_HIGH; 
171                  break; 
172              case R.id.radButton2: 
173                  priority = PRIORITY_MEDIUM; 
174                  break; 
175              case R.id.radButton3: 
176                  priority = PRIORITY_LOW; 
177          } 
178          return priority; 
179      } 
180   
181      /** 
182       * setPriority is called when we receive a task from MainActivity 
183       * 
184       * @param priority the priority value 
185       */ 
186      public void setPriorityInViews(int priority) { 
187          switch (priority) { 
188              case PRIORITY_HIGH: 
189                  ((RadioGroup) findViewById(R.id.radioGroup)).check(R.id.radButton1); 
190                  break; 
191              case PRIORITY_MEDIUM: 
192                  ((RadioGroup) findViewById(R.id.radioGroup)).check(R.id.radButton2); 
193                  break; 
194              case PRIORITY_LOW: 
195                  ((RadioGroup) findViewById(R.id.radioGroup)).check(R.id.radButton3); 
196          } 
197      } 
198  }
